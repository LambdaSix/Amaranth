// death molds.

// Resistance types
//  resists =  .air  .earth  .fire  .water    .metal  .wood    .acid  .cold  .lightning  .poison    .dark  .light    .anima  .death

:: spider resistances
    resists =  +air ++earth --fire  -water    .metal  .wood    .acid  .cold  .lightning ++poison    *dark  .light    .anima  -death

:: bat resistances
    resists =  -air  .earth  -fire  -water    -metal  -wood    .acid  -cold  -lightning  +poison    *dark  .light    .anima  -death

:: bird resistances
    resists =  -air  .earth  -fire  -water    -metal  -wood    .acid  -cold  -lightning  .poison    .dark  *light    .anima  -death

:: canine resistances
    resists =  .air  .earth  .fire  .water    .metal  .wood    .acid  -cold  .lightning  .poison    .dark  *light    .anima  -death

:: bug resistances
    resists =  +air  .earth --fire  +water    -metal  -wood    +acid  .cold  .lightning  +poison    +dark  +light    .anima  -death

:: flying insect resistances
    resists = --air  .earth --fire  +water    -metal  -wood    +acid  .cold  .lightning  +poison    .dark  *light    .anima  -death

:: dwarf resistances
    resists =  .air  +earth  -fire --water    +metal  .wood    .acid  +cold  +lightning  +poison    +dark  +light    .anima  -death

:: jelly resistances
    resists = ++air  -earth --fire --water    +metal  +wood    -acid --cold  -lightning  +poison    +dark  .light    .anima  .death
    
:: plant resistances
    resists =  +air ++earth --fire ++water    -metal  +wood    -acid  .cold  -lightning  *poison    -dark  *light    .anima  -death

// undead

:: ghost resistances
    resists = ++air ++earth  -fire ++water    *metal  *wood   ++acid ++cold  -lightning  *poison   ++dark --light   --anima  +death

:: zombie resistances
    resists =  .air  .earth ++fire  .water    -metal  -wood    -acid  *cold  +lightning  *poison    *dark --light    +anima  *death

// elementals
:: acid resistances
    resists =  +air  -earth  +fire --water    +metal  +wood    *acid  -cold  -lightning  +poison    .dark  *light    .anima  .death

:: anima resistances
    resists =  .air  .earth  .fire  .water    +metal  +wood    .acid  .cold  .lightning  -poison    .dark  *light    *anima --death

:: cold resistances
    resists =  +air  +earth --fire  +water    +metal  +wood    +acid  *cold  -lightning  +poison    +dark  -light    -anima  .death

:: dark resistances
    resists =  .air ++earth --fire  .water    +metal  +wood    +acid  +cold --lightning  +poison    *dark --light    .anima  +death

:: death resistances
    resists =  +air ++earth  +fire  -water    +metal  +wood    +acid  +cold  -lightning  +poison    *dark --light   --anima  *death
    
:: earth resistances
    resists = ++air  *earth ++fire --water    +metal  +wood    -acid  +cold  +lightning  +poison    .dark  *light    .anima  .death

:: fire resistances
    resists = ++air  -earth  *fire --water    +metal  +wood    +acid --cold  +lightning  +poison    -dark  *light    +anima  .death

:: light resistances
    resists =  .air --earth ++fire  .water    +metal  +wood    +acid  +cold ++lightning  +poison   --dark  *light    .anima  .death

:: lightning resistances
    resists = ++air --earth ++fire --water   ++metal  +wood    +acid  +cold  *lightning  +poison    -dark  +light    .anima  .death
    
:: metal resistances
    resists =  +air  +earth  +fire  +water   ++metal  *wood   --acid  +cold --lightning  +poison    .dark  *light    +anima  .death

:: poison resistances
    resists =  +air  -earth  .fire  +water    +metal  +wood   --acid  -cold  -lightning  *poison    .dark  *light   ++anima  .death
    
:: water resistances
    resists =  +air  +earth ++fire  *water    +metal  +wood    -acid --cold --lightning  -poison    .dark  *light    .anima  .death
